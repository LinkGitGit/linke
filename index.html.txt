    <script>
        // 1. 获取并格式化当前北京时间的日期（用于判断是否跨天）
        function getBeijingDate() {
            const now = new Date();
            // 强制转换为东八区（北京时间）的日期字符串，格式如：2026-04-14
            return new Intl.DateTimeFormat('zh-CN', {
                timeZone: 'Asia/Shanghai',
                year: 'numeric',
                month: '2-digit',
                day: '2-digit'
            }).format(now).replace(/\//g, '-');
        }

        // 2. 扫码次数统计逻辑
        function updateScanCount() {
            const today = getBeijingDate();
            let storedDate = localStorage.getItem('lastScanDate');
            let count = localStorage.getItem('todayScanCount');

            if (storedDate === today) {
                // 如果是今天，次数 +1
                count = parseInt(count || 0) + 1;
            } else {
                // 如果不是今天（或者第一次扫码），重置为 1
                count = 1;
                localStorage.setItem('lastScanDate', today);
            }

            // 保存最新次数到本地
            localStorage.setItem('todayScanCount', count);
            
            // 更新到页面中间
            document.getElementById('scanCount').innerText = count;
        }

        // 3. 更新屏幕下方的北京时间（精确到秒）
        function updateBeijingTime() {
            const now = new Date();
            const timeString = new Intl.DateTimeFormat('zh-CN', {
                timeZone: 'Asia/Shanghai',
                year: 'numeric',
                month: '2-digit',
                day: '2-digit',
                hour: '2-digit',
                minute: '2-digit',
                second: '2-digit',
                hour12: false // 24小时制
            }).format(now);
            
            document.getElementById('bjTime').innerText = `北京时间：${timeString}`;
        }

        // 初始化执行
        updateScanCount(); // 记录并显示次数
        updateBeijingTime(); // 立即显示一次时间
        setInterval(updateBeijingTime, 1000); // 每秒刷新一次时间
    </script>
